    网页点击特效

    # 参数：

    isColor||content

    # 类型：

    Array||Array

    # 默认值：

    12 组颜色元素的数组||12 组文字内容的数组

    点击查看代码 
    <template>
</template>
<script>
export default {
  name: "special",
  methods: {
    clickFn(e) {
      let special = document.createElement("div");
      document.body.appendChild(special);
      special.innerHTML = this.content[this.num];
      special.style.color = this.isColor[this.num];
      special.style.top = e.pageY + "px";
      special.style.left = e.pageX + "px";
      special.id = "special" + [this.num];
      setTimeout(() => {
        special.style.transition = "all 0.5s";
        special.style.transform = "translateY(-30px)";
        special.addEventListener("transitionend", () => {
          special.parentNode.removeChild(special);
        });
      }, 0);
      console.log(this.element);
      this.num++;
      if (this.num >= this.isColor.length) this.num = 0;
    },
  },
  data() {
    return {
      num: 0,
      isColor: [
        "red",
        "royalblue",
      ],
      content: [
        "123123",
        "123123",
      ],
    };
  },
  mounted() {
    document.documentElement.addEventListener("click", (e) => {
      this.clickFn(e);
    });
  },
};
</script>

    轮播图

    # 参数：

    timeout || indicatorItemSize || SwipeWidth
    SwipeHeight || imgs ||swipeLink

    # 说明：

    播放间隔，可选 || 指示器大小，可选 || 轮播图大小，可选
    轮播图大小，可选 || 图片，必选 || 图片链接，可选

    # 类型：

    Number || Number || String
    String || Array ||Array

    # 默认值:

    2000 || 12 || 375
    175 || 4*imgUrl ||#

    样式

    点击查看代码 
    <template>
  <div id="swipe" :style="{width:SwipeWidth,height:SwipeHeight}" ref="swipe">
    <div id="swipeBox" :style="{
    width:(imgs.length+2)*100+'%',
    transition:Transition,
    left:Left+'px'
    }" @transitionend="transitionend" @touchstart="touchstart" @touchmove="touchmove" @touchend="touchend">

      <div class="swipe-item">
        <img :src="imgs[imgs.length-1]" alt="img" :style="{
          width:SwipeWidth,
          height:SwipeHeight
        }">
      </div>

      <div class="swipe-item" v-for="(item,index) in imgs" :key="index+'item'">
        <img :src="item" alt="img" :style="{
          width:SwipeWidth,
          height:SwipeHeight
        }" @click="Link(index)">
      </div>

      <div class="swipe-item">
        <img :src="imgs[0]" alt="img" :style="{
          width:SwipeHeight,
          height:SwipeHeight
        }">
      </div>
    </div>
    <div id="indicator" :style="{
      width:(imgs.length*2)*indicatorItemSize+'px'
      }">
      <div v-for="(item,index) in imgs" :key="index" class="indicator-item" :class="{active:isActive===index}" :style="{
          width:indicatorItemSize+'px',
          height:indicatorItemSize+'px',
          borderRadius:indicatorItemSize+'px'}">
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: "swipe",
  data() {
    return {
      flag: true,
      Left: 0,
      sign: 1,
      Transition: "all .3s",
      timer: "",
      startX: 0,
      moveX: 0,
      isActive: 0,
      swipeWidth: 0,
      swipeHeight: 0,
      imgs: [
        require("../public/swipe/1.png"),
        require("../public/swipe/2.png"),
        require("../public/swipe/3.png"),
        require("../public/swipe/4.png"),
      ],
    };
  },
  props: {
    timeout: {
      type: Number,
      default() {
        return 2000;
      },
    },
    indicatorItemSize: {
      type: Number,
      default() {
        return 12;
      },
    },
    SwipeWidth: {
      type: String,
      default() {
        return 375 + "px";
      },
    },
    SwipeHeight: {
      type: String,
      default() {
        return 195 + "px";
      },
    },
    img: {
      type: Array,
    },
    swipeLink: {
      type: Array,
    },
  },
  watch: {
    img(n) {
      this.$nextTick(() => {
        this.imgs = n; //解决异步数据问题
      });
    },
  },
  methods: {
    // eslint-disable-next-line vue/no-dupe-keys
    Link(i) {
      if (this.swipeLink == undefined) {
        open("#", "_self");
      } else {
        open(this.swipeLink[i], "_self");
      }
    },
    Swipe() {
      //指示器
      this.isActive++;
      this.isActive == this.imgs.length ? (this.isActive = 0) : this.isActive;
      this.flag = true;
      this.sign++;
      //从第三张图开始执行过渡
      this.Left = -this.sign * this.swipeWidth;
      //每次平移一张图的左偏移
      this.Transition = "all .3s";
      //当从第一张图片开始时、打开过渡
    },
    transitionend() {
      //过渡事件
      if (this.sign >= this.imgs.length) {
        //当轮播图播到最后一个图片时
        this.sign = 0;
        //重置记录
        this.Transition = "none";
        //并且及时清除过渡
        this.Left = 0;
        //瞬间回到第二张图片
      } else if (this.sign <= 0) {
        //瞬间回到最后一张图
        this.sign = this.imgs.length;
      }
    },
    touchstart(e) {
      //点击时清除定时器预过渡
      this.startX = e.changedTouches[0].pageX;
      this.Transition = "none";
      clearInterval(this.timer);
      this.flag = false;
    },
    touchmove(e) {
      //移动时移动距离
      this.moveX = e.changedTouches[0].pageX - this.startX;
      this.Left = -this.sign * this.swipeWidth + this.moveX;
    },
    touchend() {
      //离开触摸时判断是左滑动还是有滑动
      //滑动的距离大于0为有做滑动
      if (Math.abs(this.moveX) > 50) {
        if (this.moveX > 0) {
          this.isActive--;
          this.isActive < 0
            ? (this.isActive = this.imgs.length - 1)
            : this.isActive;
          this.sign--;
        } else {
          this.isActive++;
          this.isActive == this.imgs.length
            ? (this.isActive = 0)
            : this.isActive;
          this.sign++;
        }
        //打开过渡，切换图片
        this.Transition = "all .3s";
        this.Left = -this.sign * this.swipeWidth;
      } else {
        //若果活动距离过小，不切换
        this.Transition = "all .1s";
        this.Left = -this.sign * this.swipeWidth;
      }
      //当2s之内没有触摸事件时，打开自动播放
      this.timer = setInterval(() => {
        this.Swipe();
      }, this.timeout);
    },
    init() {
      this.swipeWidth = this.$refs.swipe.offsetWidth;
      this.Left = -this.swipeWidth;
      if (this.img !== undefined) {
        this.imgs = this.img;
      }
    },
  },
  mounted() {
    //自动播放
    this.init();
    this.timer = setInterval(() => {
      this.Swipe();
    }, this.timeout);
  },
  beforeUpdate() {
    //修复当滑块快速滑动时产生数据异常的BUG
    if (!this.flag) {
      if (this.moveX > 0) {
        if (this.sign == 0) {
          this.sign = this.imgs.length;
        }
      }
      if (this.moveX < 0) {
        if (this.sign == this.imgs.length) {
          this.sign = 0;
        }
      }
    }
  },
};
</script>

    better-scroll 封装

    # 参数：

    updata || init || scrollHeight

    # 说明:

    由于异步数据的存在，而组件一般都是同步加载的
    这导致滑动无法在有异步数据渲染的时候准确的运行
    本组件提供了两种异步更新滚动的参数
    init 在第一次加载完成时调用刷新函数
    updata 在下拉加载更多时调用刷新
    scrollHeight 规定了滑动盒子的大小

    # 类型：

    Number || Boolean || String

    <template>
  <div class="scroll-wrapper" ref="scroll" :style="{height:scrollHeight}">
    <div class="scroll-content">
      <slot></slot>
    </div>
  </div>

<script>
import BScroll from "better-scroll";
import Vue from "vue";
export default {
  name: "scroll",
  data() {
    return {
      scroll: null,
    };
  },
  props: {
    updata: Number, //更新数据
    init: Boolean, //异步初始化
    scrollHeight: String,
  },
  watch: {
    init() {
      this.scroll.refresh();
    },
    updata() {
      this.$nextTick(() => {
        //better-scroll数据更新
        this.scroll.refresh();
      });
    },
  },
  methods: {
    initScroll() {
      let wrapper = this.$refs.scroll;
      this.scroll = new BScroll(wrapper, {
        probeType: 3,
        click: true,
        pullUpLoad: true,
        //scrollbar: true,
      });
      this.scroll.on("scroll", (position) => {
        this.$emit("contentScroll", position);
      });
      this.scroll.on("pullingUp", () => {
        alert("拉到底部了");
        this.scroll.finishPullUp();
        this.$emit("pullingUp");
      });
    },
  },
  mounted() {
    this.initScroll();

    let debounce = null;
    Vue.prototype.$bus = new Vue();//事件总线
    this.$bus.$on("sss", (/*a*/) => {
      if (debounce) clearTimeout(debounce); //防抖
      debounce = setTimeout(() => {
        //console.log(a); //参数传递
        this.scroll.refresh(); //监听事件总线的sss事件，与goodsItem联动
      }, 100);
    });
  },
};
</script>
